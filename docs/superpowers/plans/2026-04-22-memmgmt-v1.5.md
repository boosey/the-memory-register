```ts
// tests/core/paths.test.ts
import { describe, it, expect, beforeEach, afterEach } from "vitest";
import { resolveHomePaths } from "@/core/paths";
import path from "node:path";
import os from "node:os";

describe("resolveHomePaths", () => {
  const ORIGINAL = process.env.MEMMGMT_CLAUDE_HOME;
  afterEach(() => {
    if (ORIGINAL === undefined) delete process.env.MEMMGMT_CLAUDE_HOME;
    else process.env.MEMMGMT_CLAUDE_HOME = ORIGINAL;
  });

  it("defaults to os.homedir() + /.claude when no override", () => {
    delete process.env.MEMMGMT_CLAUDE_HOME;
    const p = resolveHomePaths();
    expect(p.claudeHome).toBe(path.join(os.homedir(), ".claude"));
  });

  it("honors MEMMGMT_CLAUDE_HOME env var", () => {
    process.env.MEMMGMT_CLAUDE_HOME = "/tmp/fake-claude";
    const p = resolveHomePaths();
    expect(p.claudeHome).toBe("/tmp/fake-claude");
    expect(p.projectsDir).toBe(path.join("/tmp/fake-claude", "projects"));
    expect(p.backupsDir).toBe(path.join("/tmp/fake-claude", "memmgmt-backups"));
  });

  it("explicit override wins over env var", () => {
    process.env.MEMMGMT_CLAUDE_HOME = "/tmp/from-env";
    const p = resolveHomePaths({ claudeHome: "/tmp/from-arg" });
    expect(p.claudeHome).toBe("/tmp/from-arg");
  });
});
```

- [ ] **Step 3: Implement the change**

Update `src/core/paths.ts`:

```ts
import os from "node:os";
import path from "node:path";

export interface HomePaths {
  claudeHome: string;
  projectsDir: string;
  pluginsDir: string;
  backupsDir: string;
  viewPrefsFile: string;
}

export interface ResolveOverride {
  home?: string;
  claudeHome?: string;
}

export function resolveHomePaths(override?: ResolveOverride): HomePaths {
  const claudeHome =
    override?.claudeHome
    ?? process.env.MEMMGMT_CLAUDE_HOME
    ?? path.join(override?.home ?? os.homedir(), ".claude");
  return {
    claudeHome,
    projectsDir: path.join(claudeHome, "projects"),
    pluginsDir: path.join(claudeHome, "plugins"),
    backupsDir: path.join(claudeHome, "memmgmt-backups"),
    viewPrefsFile: path.join(claudeHome, "memmgmt-settings.json"),
  };
}
```

> Note: if the existing `resolveHomePaths` already takes a `string` argument instead of an object, adapt the signature but keep the object form as the recommended shape. Ensure all existing call sites still compile.

```ts
// tests/e2e/fixtures.ts
import { test as base, expect, type Page } from "@playwright/test";

export const test = base.extend({});
export { expect };

/** Wait for the inventory view to be fully populated. */
export async function waitForInventory(page: Page) {
  await page.goto("/");
  await page.getByRole("heading", { name: /memmgmt/i }).waitFor({ state: "visible" });
  // Inventory renders five scope columns; wait for at least one artifact card.
  await page.getByText(/Global/i).first().waitFor({ state: "visible", timeout: 10_000 });
  await page.locator('[data-testid="artifact-card"], button:has-text("demo-skill")').first().waitFor({ timeout: 10_000 });
}

/** Click the first artifact card whose title matches (substring). */
export async function clickCardByTitle(page: Page, titleSubstring: string | RegExp) {
  const card = page.locator("button", { hasText: titleSubstring }).first();
  await card.click();
}

/** Switch to the graph tab. */
export async function switchToGraph(page: Page) {
  await page.getByRole("button", { name: /graph/i }).click();
  await page.locator(".react-flow__node").first().waitFor({ timeout: 10_000 });
}
```

### Task 2.4 — `03-edit-save-undo.spec.ts`

**Files:**
- Create: `tests/e2e/03-edit-save-undo.spec.ts`

- [ ] **Step 1: Test**

```ts
import { test, expect, waitForInventory, clickCardByTitle } from "./fixtures";
import fs from "node:fs/promises";
import path from "node:path";

const FIXTURE = path.resolve(process.cwd(), "tests/fixtures/sample-claude-home");
const SETTINGS_FILE = path.join(FIXTURE, "settings.json");
const BACKUPS = path.join(FIXTURE, "memmgmt-backups");

test("edit permission → diff preview → save → backup exists → undo restores", async ({ page }) => {
  const before = await fs.readFile(SETTINGS_FILE, "utf8");
  try {
    await waitForInventory(page);
    // Click the "Bash(git *)" permission card
    await clickCardByTitle(page, /Bash\(git \*\)/);

    // In the structured editor, find the value input and change it
    const input = page.locator('input[value="Bash(git *)"], textarea').first();
    await input.fill("Bash(git status)");

    // Trigger save → expect diff modal
    await page.getByRole("button", { name: /save/i }).first().click();
    await expect(page.getByText(/preview save|before|after/i).first()).toBeVisible();
    await page.getByRole("button", { name: /^save$/i }).last().click();

    // Backup directory should exist with a timestamped subdir
    const backupDirs = await fs.readdir(BACKUPS);
    expect(backupDirs.length).toBeGreaterThan(0);

    // Undo
    await page.getByRole("button", { name: /undo/i }).click();
    await page.waitForTimeout(500);
    const after = await fs.readFile(SETTINGS_FILE, "utf8");
    expect(after).toBe(before);
  } finally {
    // Always restore the fixture in case the test aborts mid-run
    await fs.writeFile(SETTINGS_FILE, before, "utf8");
    await fs.rm(BACKUPS, { recursive: true, force: true });
  }
});
```
